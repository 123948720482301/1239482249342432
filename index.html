        // Initialize Firebase

        const app = firebase.initializeApp(firebaseConfig);

        const database = firebase.database();

        const taskRef = database.ref("tasks");


        // DOM Elements

        const taskList = document.getElementById("taskList");

        const newTaskInput = document.getElementById("newTaskInput");

        const updateNotification = document.getElementById("updateNotification");


        // Fetch tasks from Firebase

        taskRef.on("value", (snapshot) => {

            const tasks = snapshot.val() || [];

            renderTasks(Object.entries(tasks));

        });


        // Add a new task

        newTaskInput.addEventListener("keypress", (e) => {

            if (e.key === "Enter" && newTaskInput.value.trim() !== "") {

                const newTask = { name: newTaskInput.value.trim(), completed: false };

                taskRef.push(newTask);

                newTaskInput.value = "";

            }

        });


        // Render tasks

        function renderTasks(tasks) {

            taskList.innerHTML = "";

            tasks.forEach(([id, task]) => {

                const li = document.createElement("li");


                const checkbox = document.createElement("input");

                checkbox.type = "checkbox";

                checkbox.checked = task.completed;

                checkbox.addEventListener("change", () => toggleTask(id, task.completed));

                li.appendChild(checkbox);


                const text = document.createTextNode(task.name);

                li.appendChild(text);


                taskList.appendChild(li);

            });

        }


        // Toggle task completion

        function toggleTask(taskId, completed) {

            taskRef.child(taskId).update({ completed: !completed });

        }


        // Check for app updates

        let appVersion = "1";

        setInterval(async () => {

            const response = await fetch("https://drive.google.com/uc?export=view&id=FILE_ID?v=" + new Date().getTime());

            const latestHTML = await response.text();

            if (!latestHTML.includes(`appVersion = "${appVersion}"`)) {

                updateNotification.style.display = "block";

            }

        }, 60000);


        // Refresh page

        function refreshPage() {

            window.location.reload();

        }
